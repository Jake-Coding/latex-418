\begin{problem}[104]
	Recall that the category of directed graphs is the category of presheaves $[\G^\op, \Set]$ with $\G = % https://q.uiver.app/#q=WzAsMixbMCwwLCIwIl0sWzEsMCwiMSJdLFswLDEsIlxcdGF1IiwxLHsib2Zmc2V0IjoyfV0sWzAsMSwiXFxzaWdtYSIsMSx7Im9mZnNldCI6LTJ9XV0=
	\begin{tikzcd}[cramped]
		0 & 1
		\arrow["\tau"{description}, shift right=1, from=1-1, to=1-2]
		\arrow["\sigma"{description}, shift left=2, from=1-1, to=1-2]
	\end{tikzcd}$
	
	\begin{enumerate}
		\item Compute $H_0$ and $H_1$ explicitly. Make sure they make sense.
		\item Compute $H_1 \times H_1$. Is it a square?
	\end{enumerate}
\end{problem}


\begin{solution}
	$H_\bullet: \G \to [\G^\op, \Set]$, so $H_0: \G^\op \to \Set$ is given by $H_0(-) = \G(-, 0)$. 
	
	In particular, $H_0(0) = \{1_0\}, H_0(1) = \emptyset$. Thus, because $H_0(0 \to 1)$ has an empty set as its domain (remember to op!),  $H_0(\sigma)$ is the empty function, as is $H_0(\tau)$. This "looks like" a lone vertex unconnected to anything.
	
	Similarly, $H_1(-) = \G(-, 1)$. Then, $H_1(0) = \{\sigma, \tau\}, H_1(1) = \{1_1\}, H_1(\sigma) : \G(1, 1) \to \G(0, 1)$ must be given by prescomposition by $\sigma$, and similar for $\tau$. In particular, because the only morphism in the domain is $1_1$, $H_1(\sigma)(1_1) = \sigma$, and $H_1(\tau)(1_1) = \tau$. This "looks like" a lone edge with a unique source and target.
	
	As we've discussed, $H_1 \times H_1$ as a product (hence limit) can be built from its limit (product) on its components. 
	
	We write $H_1(0) \times H_1(0)$ as a cartesian product as $\{\sigma, \tau\} \times \{\sigma, \tau\} = $\[\{(\sigma, \tau), (\tau, \sigma), (\sigma, \sigma), (\tau, \tau)\}\]
	
	We interpret this as the set of vertices of our graph.
	
	We also write $H_1(1) \times H_1(1) = \{1_1\} \times \{1_1\} = \{(1_1, 1_1)\}$. This is our set of edge (singular).
	
	To find where the edge starts, we take $(H_1(\sigma), H_1(\sigma))(1_1, 1_1) = (\sigma, \sigma)$. To find where the edge ends, we can do the same thing but with $\tau$ and get $(\tau, \tau)$. 
	
	% https://q.uiver.app/#q=WzAsNCxbMCwwLCJcXHNpZ21hXFxzaWdtYSJdLFsyLDAsIlxcc2lnbWFcXHRhdSJdLFswLDEsIlxcdGF1XFxzaWdtYSJdLFsyLDEsIlxcdGF1XFx0YXUiXSxbMCwzXV0=
	\[\begin{tikzcd}
		{\sigma\sigma} && {\sigma\tau} \\
		{\tau\sigma} && {\tau\tau}
		\arrow[from=1-1, to=2-3]
	\end{tikzcd}\]
	
	This is not a square. 
	
	

\end{solution}





\begin{problem}[106]
	Let $F: \mc{A} \to \mc{B}$ and $G: \mc{C} \to \mc{B}$ be functors. The comma category $F/G$ has objects $(a \in \mc{A}, f: Fa \to Gc, c \in \mc{C})$. 
	
	\begin{enumerate}
		\item Finish the definition of the comma category
		\item How are the other comma categories we've defined special cases of this?
		\item Prove that the category of elements of a presheaf is a comma category
	\end{enumerate}
\end{problem}
\begin{solution}
	\begin{enumerate}
		\item We need to define morphisms and check the axioms. 
		
		A morphism between $(a, f, c)$ and $(a', f', c')$ is a pair of morphisms $(\alpha: a \to a', \gamma: c\to c')$ such that the follow diagram commutes 
		% https://q.uiver.app/#q=WzAsNCxbMCwwLCJGYSJdLFsxLDAsIkdjIl0sWzAsMSwiRmEnIl0sWzEsMSwiR2MnIl0sWzAsMSwiZiJdLFsyLDMsImYnIl0sWzAsMiwiRlxcYWxwaGEiLDJdLFsxLDMsIkdcXGdhbW1hIl1d
		\[\begin{tikzcd}
			Fa & Gc \\
			{Fa'} & {Gc'}
			\arrow["f", from=1-1, to=1-2]
			\arrow["{F\alpha}"', from=1-1, to=2-1]
			\arrow["{G\gamma}", from=1-2, to=2-2]
			\arrow["{f'}", from=2-1, to=2-2]
		\end{tikzcd}\]
		
		Composition of morphisms is given componentwise. In particular, 
		
		% https://q.uiver.app/#q=WzAsNixbMCwwLCJGYSJdLFsxLDAsIkdjIl0sWzAsMSwiRmEnIl0sWzEsMSwiR2MnIl0sWzAsMiwiRmEnJyJdLFsxLDIsIkdjJyciXSxbMCwxLCJmIl0sWzIsMywiZiciXSxbMCwyLCJGXFxhbHBoYSIsMl0sWzEsMywiR1xcZ2FtbWEiXSxbMiw0LCJGXFxhbHBoYSciLDJdLFs0LDUsImYnJyJdLFszLDUsIkdcXGdhbW1hJyJdXQ==
		\[\begin{tikzcd}
			Fa & Gc \\
			{Fa'} & {Gc'} \\
			{Fa''} & {Gc''}
			\arrow["f", from=1-1, to=1-2]
			\arrow["{F\alpha}"', from=1-1, to=2-1]
			\arrow["{G\gamma}", from=1-2, to=2-2]
			\arrow["{f'}", from=2-1, to=2-2]
			\arrow["{F\alpha'}"', from=2-1, to=3-1]
			\arrow["{G\gamma'}", from=2-2, to=3-2]
			\arrow["{f''}", from=3-1, to=3-2]
		\end{tikzcd}\]
		
		Associativity follows from adding another layer to our diagram, and identities are the pair $(1_a, 1_c)$, which obviously work. 
		
		\item Taking $\mc{C}$ to be the one object category $\{*\}$, and $G$ the functor taking $* \mapsto b \in \mc{A}$. Then, $F/G$ has objects $(a \in \mc{A}, f: Fa \to b, * \in \{*\})$ and morphisms $p: a \to a'$ such that % https://q.uiver.app/#q=WzAsNCxbMCwwLCJGYSJdLFsxLDAsImIiXSxbMCwxLCJGYSciXSxbMSwxLCJiIl0sWzAsMiwiRnAiLDJdLFswLDEsImYiLDJdLFsyLDMsImYnIl0sWzEsMywiMV9iIl1d
		\[\begin{tikzcd}
			Fa & b \\
			{Fa'} & b
			\arrow["f"', from=1-1, to=1-2]
			\arrow["Fp"', from=1-1, to=2-1]
			\arrow["{1_b}", from=1-2, to=2-2]
			\arrow["{f'}", from=2-1, to=2-2]
		\end{tikzcd}\] commutes. 
		
		There's only one choice for $*$, and so we reduce our triple to a double $(a, f)$. This is precisely the comma category $F/b$. 
		
		
		
		
		Let $\mc{B} = \mc{A}, \mc{C} = \{*\}, F = 1_\mc{A}$ and $G$ the functor taking $* \mapsto b \in \mc{A}$. Then $F/G$ has objects $(a \in \mc{A}, f: a \to b, * \in \{*\})$ and morphisms $g: a \to a'$ such that 
		% https://q.uiver.app/#q=WzAsNCxbMCwwLCJGYSJdLFsxLDAsImIiXSxbMCwxLCJGYSciXSxbMSwxLCJiIl0sWzAsMl0sWzAsMSwiZiIsMl0sWzIsMywiZiciXSxbMSwzLCIxX2IiXV0=
		\[\begin{tikzcd}
			a & b \\
			{a'} & b
			\arrow["f"', from=1-1, to=1-2]
			\arrow["g"', from=1-1, to=2-1]
			\arrow["{1_b}", from=1-2, to=2-2]
			\arrow["{f'}", from=2-1, to=2-2]
		\end{tikzcd}\]
		commutes. This is the definition of the slice category $\mc{A} / b$. 
		
		\newcommand{\el}{\text{el}}
		\item Let $X: \G^\op \to \Set$ be a presheaf. The category of elements $\el X$ has objects $(S \in \G, x \in X(S))$. We need to figure out in what sense $x$ is a morphism, what our functors $F$ and $G$ are, and then verify that we have a comma category. 
		
		
		If $\mc{A} = \{*\}, F$ the functor selecting $\{*\} \in \Set$, $\mc{B} = \Set, \mc{C} = \G^\op, G = X$, we get elements $(* \in \{*\}, f: \{*\} \to X(S), S \in \G^\op)$. However, $f$ must be a function that selects a particular $x$ in $X(S)$, so we can replace $f$ with $x$ to get the same data. Moreover, there is only one choice of $*$, so we can forget it. Then, swapping the order of data in the parenthesis, we get the same data captured in $(S \in \G, x \in X(S))$. 
		
		Moreover, a morphism $(S, x) \to (S', x')$ is a morphism $g: S \to S'$ such that % https://q.uiver.app/#q=WzAsNCxbMCwwLCJcXHsqXFx9Il0sWzEsMCwiWFMiXSxbMCwxLCJcXHsqXFx9Il0sWzEsMSwiWFMnIl0sWzAsMiwiMV97XFx7KlxcfX0iLDJdLFswLDEsImYiLDJdLFsyLDMsImYnIl0sWzEsMywiWGciXV0=
		\[\begin{tikzcd}
			{\{*\}} & XS \\
			{\{*\}} & {XS'}
			\arrow["f"', from=1-1, to=1-2]
			\arrow["{1_{\{*\}}}"', from=1-1, to=2-1]
			\arrow["Xg", from=2-2, to=1-2]
			\arrow["{f'}", from=2-1, to=2-2]
		\end{tikzcd}\] commutes, where $f$ is the morphism selecting $x$ and $f'$ the morphism selecting $x'$. The $Xg$ is flipped because we want to think in $\G$, but our morphisms are from $\G^\op$. Hence, we must have that $Xg \circ f' = f$. Plugging in $*$, we get that $Xg(x') = x$. This is precisely the category of elements of $X$!
		
		That is, the comma category $(* \mapsto \{*\}) / X$ is the same as $\el X$. 
	\end{enumerate}
	
	
	
	




\end{solution}
